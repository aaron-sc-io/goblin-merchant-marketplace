import { Meta, Story, Canvas, ArgsTable, Description } from '@storybook/addon-docs/blocks';
import { useFormik } from 'formik';
import EquipmentCard from './EquipmentCard';
import { Grid, TextField, Box, MenuItem } from '@material-ui/core';
const rarities = [
  {
    value: 'Uncommon',
    label: 'Uncommon',
  },
  {
    value: 'Rare',
    label: 'Rare',
  },
  {
    value: 'Epic',
    label: 'Epic',
  },
  {
    value: 'Legendary',
    label: 'Legendary',
  },
  {
    value: 'Unique',
    label: 'Unique',
  }
];

export const Template = (props) => {
  const formik = useFormik({
    initialValues: {
      weaponType: 'Dagger',
      weaponName: 'Example Dagger',
      rarity: 'Legendary',
      weaponBaseStats: {
        weaponDmg: 30,
        weaponMoveSpd: -10
      },
      weaponBonusStats: {
        bonusPhysDmg: 2,
        bonusMgcDmg: 2,
        bonusMoveSpd: 2
      },
      weaponSlotType: 'Secondary Weapon',
      weaponHandType: 'One-Handed'
    }
  });
  const RaritySelector = ({ formik }) => {
  const handleChange = (e) => {
    formik.setFieldValue('rarity', e.target.value);
  };
  return (
    <Box width='300px' pb={2}>
      <TextField
          id="formik-textfield-rarity"
          select
          label="Rarity"
          value={formik.values.rarity}
          onChange={handleChange}
          fullWidth
          variant="outlined"
        >
          {rarities.map((option) => (
            <MenuItem key={option.value} value={option.value}>
              {option.label}
            </MenuItem>
          ))}
      </TextField>
    </Box>
  );
};
  return (
    <Grid container>
      <Grid item>
        <EquipmentCard
          formik={formik}
        />
      </Grid>
      <RaritySelector formik={formik} />
    </Grid>
  )
};

<Meta
  title='Components/EquipmentCard'
  component={EquipmentCard}
/>
